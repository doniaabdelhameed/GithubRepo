int photoresistor = A0;
int buzzer=7;
void setup()
{
Serial.begin (9600);
pinMode(buzzer,OUTPUT);
}
void loop()
{
int limit;
  limit = analogRead (photoresistor);
  Serial.println (limit);
  delay (500);

  if (limit>500)
  { digitalWrite (buzzer,HIGH);}  
  else
  { digitalWrite (buzzer,LOW);}
  }
