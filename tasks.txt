data = pd.read_csv('/content/weatherHistory.csv')

print(data.columns)

numeric_data = data.select_dtypes(include=[np.number])

features = None
target = None

if 'Temperature (C)' in numeric_data.columns:
    features = numeric_data.drop('Temperature (C)', axis=1)
    target = numeric_data['Temperature (C)']
else:
    print("The column 'Temperature (C)' does not exist in the dataset.")
    exit()

if features is not None and target is not None:
    print("Features shape:", features.shape)
    print("Target shape:", target.shape)

    print("Missing values in features:", features.isnull().sum().any())
    print("Missing values in target:", target.isnull().sum().any())

    features = features.dropna()
